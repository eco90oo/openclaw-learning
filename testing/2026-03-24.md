## 1️⃣ 基礎概念（用生活比喻，越詳細越好）

### 🎯 什麼是 Fixture？

想象你在餐廳吃飯：

**沒有 Fixture 的情況**：
- 每次上菜前，廚師都要重新洗菜、切菜、備料
- 每次都要重新熱鍋、倒油
- 效率非常低！

**有 Fixture 的情況**：
- 廚房裡已經**事先準備好**各種食材和調味料
- 每次做菜時，直接取用現成的材料
- 大幅提升效率！

> **Fixture** 就是 pytest 中的「事前準備好」的機制。它會在測試執行前準備好測試所需的數據、資源、對象，讓測試可以直接使用，不需要每次都重複準備。

### 🎯 什麼是 Parametrize？

想象你在測試一台自動販賣機：

**沒有 Parametrize 的情況**：
```
測試 1：投入 10 元，選可口可樂 → 應該掉出可樂
測試 2：投入 10 元，選雪碧 → 應該掉出雪碧
測試 3：投入 10 元，選芬達 → 應該掉出芬達
測試 4：投入 20 元，選可口可樂 → 應該掉出可樂 + 找零 10 元
...（寫到暈倒）
```

**有 Parametrize 的情況**：
```
@test.parametrize("金額", "飲料", "預期結果")
自動化產生所有測試組合！
```

> **Parametrize** 讓你只需要定義一次測試邏輯，自動產生多組測試資料。這就像「一次輸入，多重輸出」的模板概念。

## 2️⃣ 核心知識（每行程式碼都要有中文註解！）

### 🛠️ Fixture 語法詳解

```python
import pytest

# ========== 基本 Fixture（就像餐廳的備料區）==========

# @pytest.fixture 裝飾器：告訴 pytest 這是一個 Fixture
# 它會在測試函數調用前執行，並將返回值傳遞給測試
@pytest.fixture
def 餐廳材料():
    """
    這個 Fixture 會回傳一個字典，相當於餐廳的"材料箱"
    每次有測試需要"餐廳材料"時，pytest 會自動調用這個函數
    """
    return {
        "米飯": "白飯",
        "麵條": "拉麵",
        "飲料": "可樂"
    }

# 測試函數的參數名稱"餐廳材料"會自動匹配上面的 Fixture
def test_煮飯(餐廳材料):
    # 這裡可以直接使用餐廳材料，不需要自己準備！
    assert 餐廳材料["米飯"] == "白飯"
    print(f"煮的是：{餐廳材料['米飯']}")

def test_煮麵(餐廳材料):
    assert 餐廳材料["麵條"] == "拉麵"
    print(f"煮的是：{餐廳材料['麵條']}")
```

### 🏠 Fixture 的 Scope（範圍）- 控制何時創建、何時銷毀

```python
# scope 參數控制 Fixture 的生命週期
# 就像不同規模的餐廳，準備食材的頻率不同

# scope="function"（預設）：每個測試函數都會執行一次
# 就像路邊攤，每次客人點餐都要現做
@pytest.fixture(scope="function")
def 現做料理():
    print("  → 每次測試都會執行：現做料理上菜！")
    return {"料理": "現做的"}

# scope="class"：每個測試類別執行一次
# 就像家庭餐廳，一家人在同一個情境下共享前菜
@pytest.fixture(scope="class")
def 家庭套餐():
    print("  → 每個測試類別執行一次：家庭套餐上菜！")
    return {"前菜": "沙拉", "湯": "玉米濃湯"}

# scope="module"：每個模組（.py 檔案）執行一次
# 就像自助餐餐廳，整个餐厅共用一個食材庫
@pytest.fixture(scope="module")
def 自助餐食材():
    print("  → 每個模組執行一次：自助餐食材補充！")
    return {"主菜": "牛排", "甜點": "蛋糕"}

# scope="session"：整個測試 session 只執行一次
# 就像連鎖餐飲的中央工廠，一次備料供全市分店使用
@pytest.fixture(scope="session")
def 中央工廠():
    print("  → 整個測試 session 執行一次：中央工廠出貨！")
    return {"工廠名": "美味中央工"}
```

### 🍽️ autouse=True - 自動執行的 Fixture

```python
# autouse=True：自動執行，不需要測試函數主动請求
# 就像餐廳的"免費茶水"，每桌都會自動上，不需要客人點

@pytest.fixture(autouse=True)
def 免費茶水():
    """自動為每個測試提供茶水"""
    print("  📝 自動提供：免費茶水來了！")
    yield  # 測試執行後會執行 yield 後面的清理代碼
    print("  📝 測試結束：收拾茶杯")

@pytest.fixture(autouse=True)
def 餐廳環境():
    """自動設置測試環境"""
    print("  📝 自動設置：開燈、調音樂")
    yield
    print("  📝 自動清理：關燈、關音樂")

# 這些測試不需要参数，Fixtures 會自動執行
def test_訂位():
    assert True

def test_點餐():
    assert True
```

### 🎯 Parametrize - 參數化測試

```python
# @pytest.mark.parametrize 裝飾器
# 參數1：參數名稱（字串，用逗號分隔多個參數）
# 參數2：測試資料（列表，每個元素是一組測試）

@pytest.mark.parametrize("輸入,預期", [
    (1, 2),      # 第一組測試：輸入1，預期輸出2
    (2, 4),      # 第二組測試：輸入2，預期輸出4
    (3, 6),      # 第三組測試：輸入3，預期輸出6
    (10, 20),    # 第四組測試：輸入10，預期輸出20
])
def test_乘以二(輸入, 預期):
    """測試 輸入 * 2 是否等於 預期"""
    assert 輸入 * 2 == 預期

# 多參數範例：測試自動販賣機
@pytest.mark.parametrize("金額,飲料,找零,預期結果", [
    (10, "可樂", 0, "成功"),      # 投入10元，選可樂，找零0元，結果成功
    (10, "雪碧", 0, "成功"),      # 投入10元，選雪碧，找零0元，結果成功
    (20, "可樂", 10, "成功"),     # 投入20元，選可樂，找零10元，結果成功
    (5, "可樂", 5, "金額不足"),   # 投入5元，不夠，結果金額不足
    (10, "不存在的飲料", 10, "庫存不足"),  # 選不存在的飲料
])
def test_自動販賣機(金額, 飲料, 找零, 預期結果):
    """測試自動販賣機的各種情境"""
    # 模擬自動販賣機邏輯
    飲料價格 = 10
    
    if 金額 < 飲料價格:
        結果 = "金額不足"
    elif 飲料 not in ["可樂", "雪碧", "芬達"]:
        結果 = "庫存不足"
    else:
        結果 = "成功"
        實際找零 = 金額 - 飲料價格
    
    assert 結果 == 預期結果
```

### 🔗 組合使用：Parametrize + Fixture

```python
import pytest

# 先定義一個基礎的資料庫 Fixture
@pytest.fixture
def 資料庫連線():
    """模擬資料庫連線"""
    return {"主機": "localhost", "資料庫": "test_db"}

# 使用 parametrize 為每個測試提供不同的參數
# 同時會自動使用上面的資料庫連線 Fixture
@pytest.mark.parametrize("表格名稱,預期資料列數", [
    ("users", 100),
    ("products", 50),
    ("orders", 200),
])
def test_查詢資料庫(資料庫連線, 表格名稱, 預期資料列數):
    """測試查詢不同表格的記錄數"""
    # 模擬查詢結果
    模擬記錄數 = {
        "users": 100,
        "products": 50,
        "orders": 200,
    }
    
    實際數量 = 模擬記錄數.get(表格名稱, 0)
    assert 實際數量 == 預期資料列數
    print(f"查詢 {表格名稱}，共 {實際數量} 筆記錄")
```

### 🎯 進階：ids 自訂測試名稱

```python
# ids 參數讓你可以自訂每個測試的名稱，方便閱讀測試報告
@pytest.mark.parametrize(
    "輸入,預期",                      # 參數名稱
    [(1, 2), (2, 4), (3, 6)],         # 測試資料
    ids=["小數", "中數", "大數"]       # 自訂測試名稱
)
def test_乘法(輸入, 預期):
    assert 輸入 * 2 == 預期

# 執行後顯示：
# test_乘法[小數] PASSED
# test_乘法[中數] PASSED
# test_乘法[大數] PASSED
```

### 📝 練習專案：開發一個簡易計算機

讓我們創建一個完整的練習專案來運用今天學到的知識。

```python
# 計算機.py - 讓我們建立一個簡單的計算機類別

class 簡易計算機:
    """一個支援基本運算的計算機"""
    
    def 加法(self, a, b):
        """加法：a + b"""
        return a + b
    
    def 減法(self, a, b):
        """減法：a - b"""
        return a - b
    
    def 乘法(self, a, b):
        """乘法：a * b"""
        return a * b
    
    def 除法(self, a, b):
        """除法：a / b"""
        if b == 0:
            raise ValueError("除數不能為零！")
        return a / b
```

### 🧪 建立完整的測試檔案

```python
# test_計算機.py - 使用 Fixture 和 Parametrize 的完整測試

import pytest
from 計算機 import 簡易計算機

# ========== Fixture 練習 ==========

@pytest.fixture
def 計算機():
    """建立一個計算機實例供測試使用"""
    return 簡易計算機()

@pytest.fixture
def 測試資料():
    """準備測試用的資料組合"""
    return {
        "加法": [(1, 2, 3), (10, 20, 30), (-5, 5, 0)],
        "減法": [(5, 3, 2), (10, 5, 5), (0, 5, -5)],
        "乘法": [(3, 4, 12), (2, 5, 10), (-2, 3, -6)],
        "除法": [(10, 2, 5), (20, 4, 5), (100, 10, 10)],
    }

# ========== Parametrize 練習 ==========

# 測試加法 - 使用 parametrize 測試多種情況
@pytest.mark.parametrize("a,b,預期", [
    (1, 2, 3),
    (10, 20, 30),
    (-5, 5, 0),
    (0, 0, 0),
    (100, 200, 300),
])
def test_加法(計算機, a, b, 預期):
    """測試加法功能"""
    結果 = 計算機.加法(a, b)
    assert 結果 == 預期, f"期望 {a} + {b} = {預期}，但得到 {結果}"

# 測試減法
@pytest.mark.parametrize("a,b,預期", [
    (5, 3, 2),
    (10, 5, 5),
    (0, 5, -5),
    (-10, -5, -5),
])
def test_減法(計算機, a, b, 預期):
    """測試減法功能"""
    結果 = 計算機.減法(a, b)
    assert 結果 == 預期

# 測試乘法
@pytest.mark.parametrize("a,b,預期", [
    (3, 4, 12),
    (2, 5, 10),
    (-2, 3, -6),
    (0, 100, 0),
    (7, 8, 56),
])
def test_乘法(計算機, a, b, 預期):
    """測試乘法功能"""
    結果 = 計算機.乘法(a, b)
    assert 結果 == 預期

# 測試除法 - 正常情況
@pytest.mark.parametrize("a,b,預期", [
    (10, 2, 5.0),
    (20, 4, 5.0),
    (100, 10, 10.0),
    (9, 3, 3.0),
])
def test_除法正常(計算機, a, b, 預期):
    """測試除法 - 正常情況"""
    結果 = 計算機.除法(a, b)
    assert 結果 == 預期

# 測試除法 - 例外情況
def test_除法例外_除數為零(計算機):
    """測試除法 - 除數為零應該拋出例外"""
    with pytest.raises(ValueError) as 例外資訊:
        計算機.除法(10, 0)
    
    assert "除數不能為零" in str(例外資訊.value)

# ========== Fixture Scope 練習 ==========

@pytest.fixture(scope="class")
def 計算機記錄器():
    """記錄所有計算機操作的日誌"""
    記錄 = []
    yield 記錄  # yield 讓我們在測試後還能訪問這個列表
    # 測試結束後顯示記錄
    print(f"\n📊 計算機操作記錄：{記錄}")

class Test計算機整合測試:
    """整合測試類別"""
    
    def test_連續運算(self, 計算機, 計算機記錄器):
        """測試連續進行多個運算"""
        # 紀錄操作
        計算機記錄器.append("加法: 5 + 3 = 8")
        assert 計算機.加法(5, 3) == 8
        
        計算機記錄器.append("乘法: 8 * 2 = 16")
        assert 計算機.乘法(8, 2) == 16
        
        計算機記錄器.append("減法: 16 - 4 = 12")
        assert 計算機.減法(16, 4) == 12
    
    def test_計算結果正確(self, 計算機):
        """驗證最終結果"""
        # (5 + 3) * 2 - 4 = 12
        result = 計算機.減法(
            計算機.乘法(
                計算機.加法(5, 3),
                2
            ),
            4
        )
        assert result == 12
```
